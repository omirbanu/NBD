Generating an p-ER random graph

01 Create an empty Graph
02 Create empty list edges
03 Add n nodes into the Graph
04 for each node source in Graph:
05 for each node target in Graph:
06 pick a random number rp from 0 to 1
07 if source!=target and (source,target) not in edges and (target,source) not in edges
08 if  rp<=probability p
09 add edge to the list and to the graph
10 endfor
11 endfor

def erdos_renyi_graph(n_n,pp):
    '''
    Method that generates a p-ER random graph.
    input:
        n_n - number of nodes 
        pp - probability that an edge is included in the graph; independently of any other edge
    output:
        generated erdos_renyi_graph
    '''
    our_graph=nx.Graph() # method that creates and empty graph from nx library
    edges=[] # list of edges 
    nodes=range(0,n_n-1) 
    our_graph.add_nodes_from(nodes) #adding nodes
    for source in range(n_n):
        for target in range(n_n):
            rp=random.random() # pick a random number from 0 to 1
            if source!=target and (source,target) not in edges and (target,source) not in edges: #check that source node and target node are not the same and that edge is not in edges of the graph
                if rp<=pp:#if the number is less than probability
                    edges.append((source,target))
                    our_graph.add_edge(source,target)#add that edge to the graph
    return our_graph

def edges_r_g(rg,r,n, node,nodes):
    '''
    Method that returns a node that with an input node will form an edge in process of generating r-regular random graph
    Input:
        rg - input Graph with all the nodes
        r -  The degree of each node that should be in the generated graph
        n -  The number of nodes
        node - input node
        nodes - list of all the nodes from the graph
    Output:
        sampled node that is in accordance with r degree constraint
    '''
    n_edges=set(rg.neighbors(node)) #set of neighbouring nodes 
    if len(n_edges)<r:# if the number of neigbours is smaller than r
        req_n=1#the size of the sample is 1 
    else:
        return False#"Error: r value exceeded."
    for nd in nodes:#check every node
        if len(list(rg.neighbors(nd)))>=r: #if the number of node neighbours is bigger or same as r value
            n_edges.add(nd) #store to the n_edges set
    n_edges.add(node)#add the input node in order to avoid sampling itself
    
    poss_nodes=list(set(nodes).difference(n_edges))#poss_nodes-possible nodes that should have more neigbours to have a 'r degree'
    if len(poss_nodes)==0:
        return False#"All nodes have r degree."
 
    rand_nodes=random.sample(poss_nodes,req_n)#sample req_n (sample=1); sample one node and return it  
    return (rand_nodes[0])


def random_regular_graph(r,n):
    '''
    Method that generates a r-regular random graph
    input:
        r - The degree of each node
        n - The number of nodes
    output: Generated r-regular random graph
    '''
    if ((r*n)%2!=0) or (r>n): #check if the multiplication of r and n is odd or r is bigger than n
        print ("r*n is not even or r is less than n") 
    else:
        rand_g=nx.Graph()#create an empty graph
        nodes=range(0,n)#create nodes
        rand_g.add_nodes_from(nodes)#add n created nodes
        for ri in range(r):#for every index ri in r
            for node in nodes:#for every node in list of all the nodes
                r_node=edges_r_g(rand_g,r,n,node,nodes)#return one of the possible nodes 
                if r_node !=False:#node found
                    if len(list(rand_g.neighbors(r_node)))<r:#additional checking before adding the edge between nodes 
                        rand_g.add_edge(r_node,node)
        return rand_g

A graph is said to be connected if there exists a path connecting any two nodes.

A graph is connected if one of the following statements is true

    -The adjacency matrix A is irreducible;
    -The second smallest eigenvalue of L is positive;

Method 1 - algebraic method 1 (irreducibility)¶

If we consider the algebraic method 1 (irreducibility) a graph is connected if the following statement is true

  The adjacency matrix A is irreducible;

An nxn matrix A is irreducible if

$I + A + A^2 + ... + A^{n-1} > 0$ , whereby I is an identity matrix

In other words all the elements from the matrix calculated from the expression above have to be positive. If they are all positive, we can conclude that

    The adjacency matrix A is irreducible; Therefore, the given graph is connected.

def irr_check(G):
    '''
    Method that checks the connectivity of the given graph by using irreducibility method of the matrix 
    input: given graph
    output:
          res_matrix - matrix calculated from the formula above (for checking irreducibility on the adjacency matrix)
          irreducible - boolean value; True-Matrix is irreducible-Graph is connected; False-otherwise
    '''
    N=len(G.nodes())# N - number of nodes
    A=nx.adjacency_matrix(G).todense()# A - adjacency matrix
    res_matrix=np.identity(N,dtype=float)# res_matrix for storing the sums of identity and powered adjacency matrices
    for i in range(1,N):# beacuse we need it to N-1
        res_matrix+=np.linalg.matrix_power(A,i)
    pos_vals=sum(res_matrix>0).sum()
    no_elements=res_matrix.shape[0]**2
    res_str=''
    irreducible=True
    if pos_vals==no_elements:
        res_str='Matrix is irreducible->Graph is well connected!'
        irreducible=True
    else:
        res_str='Matrix is reducible->Graph is not well connected!'
        irreducible=False
    #print(res_str)
    return res_matrix,irreducible

Method 2 - algebraic method 2 (eigenvalue of the Laplacian matrix)¶

The Laplacian matrix of a graph is an nxn matrix L defined as follows:

$L_{ii} = d_i $, where $d_i$ is the degree of node i ;

$L_{ij}= -1$ if and only if (i ; j) $\epsilon$ A.

L = D - A

Since the graphs in the task are undirected, L is a symmetric matrix.

Connectivity of a graph through algebraic method - computing eigenvalue of the Laplacian matrix¶

Laplacian matrix = Kirchhoff matrixis amatrix representation of agraph

We are going to check the connectivity of a graph through computing the second smallest eigenvalue of the Laplacian graph, if the graph is connected, then the second smallest eigenvalue is positive. (where the first smallest value is 0)

L - laplacian matrix X - non-zero column vector of n real numbers (n-dim of matrix) As xT Lx >= 0 which means L - positive semi-definite matrix, then eigenvalues of L are real and > 0.

def lapl_eig(G):
    '''
    Method that checks the connectivity of the given graph by using irreducibility method of the matrix 
    input: given graph
    output:
          res_matrix - matrix calculated from the formula above (for checking irreducibility on the adjacency matrix)
          irreducible - boolean value; True-Matrix is irreducible-Graph is connected; False-otherwise
    '''
    n=len(G.nodes())
    D=np.identity(n)
    for i in range(0,n):
        D[i,i]=G.degree(i) #Degree matrix D
    A = nx.adjacency_matrix(G)#adjacency matrix
    L = D - A#Laplacian matrix
  #  eigval,eigvec=sparse.linalg.eigsh(L,k=2,sigma=0,which='LM')# solution to 
    eigval,eigvec=sparse.linalg.eigsh(L,k=L.shape[0],sigma=0,which='LM')# solution to 

    res_str=''
    connected=True
    if eigval[1]>0:
        res_str='Well connected graph!'
        connected=True
    else:
        res_str='Not connected graph'
        connected=False
  #  print(res_str)
    return eigval,eigvec,connected

def bfs_t(G, start):
    exp=[]    #exploring nodes
    queue=[start]  #starting point
    
    #exploring nodes untill no nodes 
    while queue:
        nd=queue.pop(0)
        if nd not in exp:
            exp.append(nd)
            nbrs=G.neighbors(nd) #getting adjacent nodes

            for nr in nbrs:
                queue.append(nr)   #take for exploring 
    if (len(exp)==len(G)):
        return True
    else:
        return False

        estimated probability based on the frequency that graph was connected by checking from

    num_of_conn=0

def mc_reg(r1,n_mc):
    '''
    Method that performes Monte Carlo simulation to estimate the value of probability thath Graph 
    is connected
    input: 
        r1-degree of the nodes
        n_mc-number of iterations in MC simulation
    output:
        estimated probability based on the frequency that graph was connected by checking from 
        the BFS method.
    '''
    lst_of_probs=[]#list of the estimated probabilities
    for i in range(r1+1,100):#for every value n
        num_of_r1=0#counter for connected graphs
        for j in range(0,n_mc):#for every MC iteration
            gr=random_regular_graph(r1,i)#create random_regular_graph with r1 degree of nodes
            if bfs_t(gr,0)==True:
                num_of_r1+=1#count the connected ones
        lst_of_probs.append((i,num_of_r1/n_mc))#store estimated probabilities
    return lst_of_probs
